import { E as store_get, F as escape_html, G as attr_style, K as stringify, I as unsubscribe_stores, D as pop, z as push } from "./index.js";
import { p as progressStore } from "./progress.js";
function ProgressTracker($$payload, $$props) {
  push();
  var $$store_subs;
  let {
    lessonId,
    sectionTitle = "",
    showNavigation = true
  } = $$props;
  let progress = store_get($$store_subs ??= {}, "$progressStore", progressStore);
  let currentLesson = progress.phases.flatMap((p) => p.lessons).find((l) => l.id === lessonId);
  let isCompleted = currentLesson?.completed || false;
  let mounted = typeof window !== "undefined";
  let sectionProgress = (() => {
    if (!lessonId) return null;
    let phaseNumber;
    let sectionName;
    if (lessonId.startsWith("phase")) {
      const parts = lessonId.split("-");
      phaseNumber = parseInt(parts[0].replace("phase", ""));
      sectionName = parts[1];
    } else {
      const [phaseStr, sectionStr] = lessonId.split(".");
      phaseNumber = parseInt(phaseStr);
      sectionName = `section${sectionStr ? sectionStr[0] : "1"}`;
    }
    const currentPhase = progress.phases[phaseNumber - 1];
    if (!currentPhase) return null;
    const sectionLessons = currentPhase.lessons.filter((lesson) => {
      if (lesson.id.startsWith("phase")) {
        const parts = lesson.id.split("-");
        return parts.length > 1 && parts[1] === sectionName;
      } else {
        const [, lessonSectionStr] = lesson.id.split(".");
        return lessonSectionStr && lessonSectionStr[0] === sectionName.replace("section", "");
      }
    });
    const completedCount = sectionLessons.filter((l) => l.completed).length;
    return {
      total: sectionLessons.length,
      completed: completedCount,
      percentage: Math.round(completedCount / sectionLessons.length * 100)
    };
  })();
  if (mounted) {
    $$payload.out += "<!--[-->";
    $$payload.out += `<div class="bg-white border border-gray-200 rounded-lg p-4 my-6"><div class="flex items-center justify-between mb-4"><h3 class="text-sm font-medium text-gray-700">📊 学習進捗</h3> `;
    if (isCompleted) {
      $$payload.out += "<!--[-->";
      $$payload.out += `<div class="flex items-center text-green-600 text-sm"><svg class="w-4 h-4 mr-1" fill="currentColor" viewBox="0 0 20 20"><path fill-rule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clip-rule="evenodd"></path></svg> 完了済み</div>`;
    } else {
      $$payload.out += "<!--[!-->";
    }
    $$payload.out += `<!--]--></div> <div class="mb-4"><div class="flex items-center justify-between text-sm text-gray-600 mb-2"><span>現在のレッスン: ${escape_html(lessonId)}</span> `;
    if (currentLesson?.timeSpent) {
      $$payload.out += "<!--[-->";
      $$payload.out += `<span>学習時間: ${escape_html(currentLesson.timeSpent)}分</span>`;
    } else {
      $$payload.out += "<!--[!-->";
    }
    $$payload.out += `<!--]--></div></div> `;
    if (sectionProgress && sectionTitle) {
      $$payload.out += "<!--[-->";
      $$payload.out += `<div class="mb-4 p-3 bg-gray-50 rounded-lg"><div class="flex items-center justify-between text-sm text-gray-700 mb-2"><span>${escape_html(sectionTitle)}</span> <span>${escape_html(sectionProgress.completed)}/${escape_html(sectionProgress.total)}</span></div> <div class="progress-bar h-2"><div class="progress-fill h-2 rounded-full"${attr_style(`width: ${stringify(sectionProgress.percentage)}%`)}></div></div> <div class="text-xs text-gray-500 mt-1 text-right">${escape_html(sectionProgress.percentage)}% 完了</div></div>`;
    } else {
      $$payload.out += "<!--[!-->";
    }
    $$payload.out += `<!--]--> `;
    if (lessonId) {
      $$payload.out += "<!--[-->";
      const phaseNumber = lessonId.startsWith("phase") ? parseInt(lessonId.split("-")[0].replace("phase", "")) : parseInt(lessonId.split(".")[0]);
      const phaseProgress = progress.phases[phaseNumber - 1];
      if (phaseProgress) {
        $$payload.out += "<!--[-->";
        $$payload.out += `<div class="p-3 bg-blue-50 rounded-lg"><div class="flex items-center justify-between text-sm text-blue-700 mb-2"><span>Phase ${escape_html(phaseProgress.phase)}: ${escape_html(phaseProgress.name)}</span> <span>${escape_html(phaseProgress.completedLessons)}/${escape_html(phaseProgress.totalLessons)}</span></div> <div class="progress-bar h-2 bg-blue-200"><div class="h-2 rounded-full bg-blue-600"${attr_style(`width: ${stringify(phaseProgress.completedLessons / phaseProgress.totalLessons * 100)}%`)}></div></div> <div class="text-xs text-blue-600 mt-1 text-right">${escape_html(Math.round(phaseProgress.completedLessons / phaseProgress.totalLessons * 100))}% 完了</div></div>`;
      } else {
        $$payload.out += "<!--[!-->";
      }
      $$payload.out += `<!--]-->`;
    } else {
      $$payload.out += "<!--[!-->";
    }
    $$payload.out += `<!--]--> `;
    if (showNavigation) {
      $$payload.out += "<!--[-->";
      $$payload.out += `<div class="mt-4 pt-4 border-t border-gray-200"><div class="flex justify-between"><a href="/phase1" class="text-sm text-blue-600 hover:text-blue-800 flex items-center"><svg class="w-4 h-4 mr-1" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15 19l-7-7 7-7"></path></svg> Phase 1 概要に戻る</a> <a href="/table-of-contents" class="text-sm text-gray-600 hover:text-gray-800 flex items-center">全体目次 <svg class="w-4 h-4 ml-1" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"></path></svg></a></div></div>`;
    } else {
      $$payload.out += "<!--[!-->";
    }
    $$payload.out += `<!--]--></div>`;
  } else {
    $$payload.out += "<!--[!-->";
  }
  $$payload.out += `<!--]-->`;
  if ($$store_subs) unsubscribe_stores($$store_subs);
  pop();
}
export {
  ProgressTracker as P
};
